\begin{lemma}\label{sum_pow}
	\lean{sum_pow}
	\leanok
	
	Given (n : ℕ) {x : ℝ} (hx : x ≠ 1), we have
    ∑ i ∈ range n, x^i = (x ^ n - 1) / (x - 1).
\end{lemma}
